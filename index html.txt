    <!-- JavaScript Logic -->
    <script>
        // Data & Variables
        let currentBasePriceUSD = 0;
        let exchangeRate = 1600; // Simulated Exchange Rate: $1 = 1600 NGN

        // Navbar Scroll Effect
        window.addEventListener('scroll', function() {
            const navbar = document.getElementById('navbar');
            if (window.scrollY > 50) {
                navbar.classList.add('bg-black/90', 'backdrop-blur-md', 'border-b', 'border-gray-800', 'py-4');
                navbar.classList.remove('py-6');
            } else {
                navbar.classList.remove('bg-black/90', 'backdrop-blur-md', 'border-b', 'border-gray-800', 'py-4');
                navbar.classList.add('py-6');
            }
        });

        // Mobile Menu Toggle
        function toggleMobileMenu() {
            const menu = document.getElementById('mobileMenu');
            menu.classList.toggle('hidden');
            menu.classList.toggle('flex');
        }

        // Modal Logic
        function toggleModal(modalId) {
            const modal = document.getElementById(modalId);
            if (modal.classList.contains('hidden-modal')) {
                modal.classList.remove('hidden-modal');
                modal.classList.add('visible-modal');
            } else {
                modal.classList.add('hidden-modal');
                modal.classList.remove('visible-modal');
            }
        }

        // --- NEW: Admin Login Logic ---
        function checkAdminPin() {
            const pin = document.getElementById('adminPinInput').value;
            const errorMsg = document.getElementById('pinError');
            
            // Updated Security PIN
            if (pin === "077447") {
                toggleModal('adminLoginModal'); // Close login
                toggleModal('walletModal');     // Open wallet
                document.getElementById('adminPinInput').value = ""; // Clear Input
                errorMsg.classList.add('hidden');
            } else {
                errorMsg.classList.remove('hidden');
            }
        }

        // --- NEW: Name Verification Logic ---
        function verifyAccountName() {
            const accNum = document.getElementById('accountNumber').value;
            const display = document.getElementById('accountNameDisplay');
            
            if (accNum.length < 5) {
                alert("Please enter a valid account number first.");
                return;
            }

            display.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Checking...';
            display.classList.remove('text-gray-500', 'italic');
            display.classList.add('text-yellow-500', 'font-bold');

            // Simulate API delay
            setTimeout(() => {
                display.classList.remove('text-yellow-500');
                display.classList.add('text-green-400');
                display.innerHTML = '<i class="fa-solid fa-check-circle"></i> SHADY CREATION USER';
            }, 1500);
        }

        // Service Filtering
        function filterServices(category) {
            const cards = document.querySelectorAll('.service-card');
            const btns = document.querySelectorAll('.filter-btn');

            btns.forEach(btn => {
                if(btn.dataset.filter === category) {
                    btn.classList.remove('bg-gray-900', 'text-gray-400');
                    btn.classList.add('bg-white', 'text-black');
                } else {
                    btn.classList.add('bg-gray-900', 'text-gray-400');
                    btn.classList.remove('bg-white', 'text-black');
                }
            });

            cards.forEach(card => {
                if (category === 'all' || card.dataset.category === category) {
                    card.style.display = 'block';
                } else {
                    card.style.display = 'none';
                }
            });
        }

        // Order Logic
        function openOrderModal(title, priceUSD, unit, requiresUpload = true) {
            currentBasePriceUSD = priceUSD;
            document.getElementById('order-title').innerText = title;
            document.getElementById('order-unit').innerText = '/ ' + unit;
            
            // Reset to USD default
            setCurrency('USD');
            
            if (requiresUpload) {
                // Normal Logic (File Upload -> Payment)
                document.getElementById('order-step-1').classList.remove('hidden');
                document.getElementById('order-step-2').classList.add('hidden');
                document.getElementById('file-name').classList.add('hidden');
                document.getElementById('btn-continue').disabled = true;
                document.getElementById('btn-continue').classList.add('bg-gray-800', 'text-gray-500', 'cursor-not-allowed');
                document.getElementById('btn-continue').classList.remove('bg-yellow-500', 'text-black');
            } else {
                // Skip Upload Logic (For Site Creation/Apps)
                document.getElementById('order-step-1').classList.add('hidden');
                document.getElementById('order-step-2').classList.remove('hidden');
            }

            toggleModal('orderModal');
        }

        function closeOrderModal() {
            toggleModal('orderModal');
        }

        function fileUploaded(input) {
            if (input.files && input.files[0]) {
                const fileNameDiv = document.getElementById('file-name');
                const btn = document.getElementById('btn-continue');
                
                fileNameDiv.innerText = "File selected: " + input.files[0].name;
                fileNameDiv.classList.remove('hidden');
                
                btn.disabled = false;
                btn.classList.remove('bg-gray-800', 'text-gray-500', 'cursor-not-allowed');
                btn.classList.add('bg-yellow-500', 'text-black');
            }
        }

        function goToPayment() {
            document.getElementById('order-step-1').classList.add('hidden');
            document.getElementById('order-step-2').classList.remove('hidden');
        }

        // Currency Toggle Logic
        function setCurrency(currency) {
            const priceEl = document.getElementById('order-price');
            const displayEl = document.getElementById('price-display');
            const btnUsd = document.getElementById('btn-usd');
            const btnNgn = document.getElementById('btn-ngn');

            if (currency === 'USD') {
                priceEl.innerText = currentBasePriceUSD;
                displayEl.innerHTML = $<span id="order-price">${currentBasePriceUSD}</span>;
                
                btnUsd.classList.add('bg-yellow-500', 'text-black');
                btnUsd.classList.remove('text-gray-400');
                btnNgn.classList.remove('bg-yellow-500', 'text-black');
                btnNgn.classList.add('text-gray-400');
            } else {
                const priceNGN = (currentBasePriceUSD * exchangeRate).toLocaleString();
                displayEl.innerHTML = ₦<span id="order-price">${priceNGN}</span>;
                
                btnNgn.classList.add('bg-yellow-500', 'text-black');
                btnNgn.classList.remove('text-gray-400');
                btnUsd.classList.remove('bg-yellow-500', 'text-black');
                btnUsd.classList.add('text-gray-400');
            }
        }
    </script>
